#define MAX 10
int mang[MAX], n;

void swap (int *x, int *y)
{
  int tmp;
  tmp = *x;
  *x = *y;
  *y = tmp;
}

void hoanvi(int k)
{
  int j;

  if (k==1)
  {
    printf("\n");
    for (j=0; j<n; j++)
      printf("%d ", mang[j]);
  }
  else
    for (j=k-1; j>=0; j--)
    {
      swap(&mang[k-1], &mang[j]);
      hoanvi(k-1);
      swap(&mang[j], &mang[k-1]);
    }
}

void main()
{
  int i;

#include <stdio.h>
#include <conio.h>

unsigned long giaithua(int n)
{
   unsigned long ketqua = 1;
   int i;

   for (i=2; i<=n; i++)
     ketqua *= i;

   return ketqua;
}

unsigned long to_hop_chap(int k, int n)
{
   return giaithua(n) / (giaithua(k) * giaithua(n-k));
}

void main()
{
  int n, k;

  printf("\nNhap vao gia tri N va K : ");
  scanf("%d%d", &n, &k);

  printf("Top hop chap %d cua %d = %lu", k, n, to_hop_chap(k, n));
  getch();
}

 	int a,b,c,d;

   cout<<"Nhap vao so thu 3: ";cin>>c;

float max(int a, int b)
{
 	if(a>b)
 	return a;
 	else
 	return b;
}
float min(int a, int b)
{
 	if(a>b)
 	return b;
 	else
 	return a;
}
