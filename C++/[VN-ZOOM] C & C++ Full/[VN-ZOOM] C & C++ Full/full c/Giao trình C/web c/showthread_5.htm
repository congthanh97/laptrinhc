void qs(int *A,int left,int right)
  {
    int i,j,x,y;
    i=left;
    j=right;
    x=A[(left+right)/2];
    do
      {
	while(A[i]<x&&i<right)i++;
	while(A[j]>x&&j>left)j--;
	if(i<=j)
	  {
	    y=A[i];
	    A[i]=A[j];
	    A[j]=y;
	    i++;
	    j--;
	  }
      }while(i<=j);
    if(left<j)qs(A,left,j);
    if(i<right)qs(A,i,right);
  }

void merge(int *A,int n)
  {
    int i,k,j,low1,up1,low2,up2,size;
    int *ds;
    size=1;
    ds=(int*)malloc(n*sizeof(int));
    while(size<n)
      {
	low1=0;
	k=0 ;
	while(low1+size<n)
	  {
	    low2=low1+size;
	    up1=low2-1;
	    if(low2+size-1<n)
	      up2=low2+size-1;
	    else
	      up2=n-1;
	    for(i=low1,j=low2;i<=up1 && j<=up2;k++)
	      {
		if(A[i]<=A[j])
		  ds[k]=A[i++];
		else
		  ds[k]=A[j++];
	      }
	    for(;i<up1;k++)
	      ds[k]=A[i++];
	    for(;j<up2;k++)
	      ds[k]=A[j++];
	    low1=up2+1;
	  }
	for(i=low1;k<n;i++)
	  ds[k++]=A[i];
	for(i=0;i<n;i++)
	  A[i]=ds[i];
	size*=2;
      }

    free(ds);
